const state = { data:{}, quotes:{}, lastQuoteFetch:0 };


function parseCSV(text){

  const rows=[]; let row=[], cur="", q=false;

  for(let i=0;i<text.length;i++){

    const ch=text[i];

    if(q){ if(ch==='"'){ if(text[i+1]==='"'){cur+='"';i++;} else q=false; } else cur+=ch; }

    else{

      if(ch==='"') q=true;

      else if(ch===","){ row.push(cur); cur=""; }

      else if(ch==="\n"){ row.push(cur); rows.push(row); row=[]; cur=""; }

      else if(ch!=="\r") cur+=ch;

    }

  }

  if(cur.length||row.length){ row.push(cur); rows.push(row); }

  return rows;

}

function parseNum(s){

  if(s==null) return null;

  const c=String(s).replace(/[$,฿\s]/g,"");

  if(c==="") return null;

  const n=parseFloat(c);

  return Number.isFinite(n)?n:null;

}

// รับเฉพาะตัวเลขล้วน (ปฏิเสธ "193 / $15", "JUL" ฯลฯ) สำหรับเงื่อนไข eyes

function parseStrictNum(s){

  if(s==null) return null;

  const c=String(s).trim().replace(/[$,฿\s]/g,"");

  if(c==="") return null;

  if(!/^-?\d*\.?\d+$/.test(c)) return null;

  const n=parseFloat(c);

  return Number.isFinite(n)?n:null;

}

// ถ้า note เป็นตัวเลขล้วน (อาจมี comma คั่นหลักพัน) เอา comma ออก; ถ้าเป็น text อื่นคงไว้

function stripThousands(s){

  if(/^-?[\d,]+(\.\d+)?%?$/.test(s) && s.includes(",")) return s.replace(/,/g,"");

  return s;

}

function extractStocks(rows){

  if(!rows||rows.length<3) return [];

  const out=[]; const width=Math.max(...rows.slice(0,9).map(r=>r.length));

  for(let col=1;col<width;col++){

    const ticker=(rows[0]&&rows[0][col]||"").trim();

    if(!ticker) continue;

    out.push({

      ticker,

      symbol:ticker.replace(/\*/g,"").trim().toUpperCase(),

      note:stripThousands((rows[1]&&rows[1][col]||"").trim()),

      sheetPrice:parseNum(rows[2]&&rows[2][col]),

      t:[parseNum(rows[3]&&rows[3][col]),parseNum(rows[4]&&rows[4][col]),

         parseNum(rows[5]&&rows[5][col]),parseNum(rows[6]&&rows[6][col])],

      row8:parseStrictNum(rows[7]&&rows[7][col]),

      row9:parseStrictNum(rows[8]&&rows[8][col]),

    });

  }

  return out;

}

function priceOf(s){

  const q=state.quotes[s.symbol];

  if(q!=null) return {value:q, live:true};

  return {value:s.sheetPrice, live:false};

}

// scheme ของหุ้นตัวนั้น (อาจถูก override เป็น "none" ด้วย noZoneFrom)

function schemeFor(s,tab){ return SCHEMES[s.scheme||tab.scheme]; }

function colorFor(s,tab){

  const sch=schemeFor(s,tab);

  if(sch.hideZones) return COLORS.darkgrey; // tier 3 (buy red 11+) พื้นเข้มให้ 🧠/🧀 เด่น

  const i=sch.band(priceOf(s).value,s.t);

  if(i==null) return COLORS.plain;

  if(tab.zoneMaxBand!=null && i>tab.zoneMaxBand) return COLORS.plain; // band ตื้นเกิน -> เทา

  return COLORS[sch.palette[i]];

}

function fmtPrice(n){

  if(n==null) return "—";

  return "$"+(Math.round(n*100)/100);

}

function esc(s){ return s.replace(/[&<>"']/g, m=>({"&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;","'":"&#39;"}[m])); }


async function loadTab(tab){

  try{

    const url=PUB_BASE+"?gid="+tab.gid+"&single=true&output=csv&cachebust="+Date.now();

    const res=await fetch(url,{cache:"no-store"});

    if(!res.ok) throw new Error("HTTP "+res.status);

    const text=await res.text();

    if(text.trim().startsWith("<")) throw new Error("ได้ HTML แทน CSV — เช็คว่า publish เป็น CSV แล้ว");

    const stocks=extractStocks(parseCSV(text));

    if(!stocks.length) throw new Error("ไม่พบข้อมูลหุ้น");

    stocks.forEach((s,i)=>{

      s.scheme = (tab.noZoneFrom && (i+1)>=tab.noZoneFrom) ? "none" : tab.scheme;

    });

    state.data[tab.id]={stocks};

  }catch(e){

    state.data[tab.id]={...(state.data[tab.id]||{}), error:e.message||"โหลดไม่สำเร็จ"};

  }

}


async function fetchQuote(symbol){

  try{

    const res=await fetch("https://finnhub.io/api/v1/quote?symbol="+encodeURIComponent(symbol)+"&token="+FINNHUB_KEY);

    if(!res.ok) return;

    const j=await res.json();

    if(j&&typeof j.c==="number"&&j.c>0) state.quotes[symbol]=j.c;

  }catch(e){ /* fallback ราคา Sheet */ }

}


async function fetchAllQuotes(){

  const now=Date.now();

  if(now-state.lastQuoteFetch<QUOTE_COOLDOWN_MS) return;

  state.lastQuoteFetch=now;

  const symbols=new Set();

  TABS.forEach(tab=>{

    const d=state.data[tab.id]||{};

    (d.stocks||[]).forEach(s=>symbols.add(s.symbol));

  });

  await Promise.all([...symbols].map(fetchQuote));

}


async function loadAll(){

  const btn=document.getElementById("refresh");

  btn.classList.add("spin");

  if(!TABS.some(t=>(state.data[t.id]||{}).stocks)){

    document.getElementById("grid-area").innerHTML='<div class="msg">กำลังดึงข้อมูล...</div>';

  }

  await Promise.all(TABS.map(loadTab));

  render();

  await fetchAllQuotes();

  render();

  btn.classList.remove("spin");

}


function render(){

      area.appendChild(div);

    }

  TABS.forEach(tab=>{

    const d=state.data[tab.id]||{};

    if(!d.stocks) return;

    d.stocks.forEach(s=>{

      const btn=document.createElement("button");

      const pv = priceOf(s).value;

      const eyes = (s.row8!=null && pv!=null && pv <= s.row8) ? "🧠" : "";

      btn.onclick=()=>openDetail(s,tab);

      grid.appendChild(btn);

    });

  });

  if(grid.children.length) area.appendChild(grid);

}


function openDetail(s,tab){

  const sch=schemeFor(s,tab);

  const c=colorFor(s,tab);

  const p=priceOf(s);

  // ใช้ scheme "deep" คำนวณโซนใน popup เสมอถ้า threshold ครบ (แม้ tile จะเป็น none/เทา)

  const zsch = sch.hideZones ? SCHEMES.deep : sch;

  const idx=zsch.band(p.value,s.t);

  const hasZones = s.t.every(x=>x!=null);

  let rows="";

  if(hasZones){

    s.t.forEach((th,i)=>{

      const band=zsch.thBand(i);

      const on=idx!=null&&idx===band;

      const bc=COLORS[zsch.palette[band]];

      rows+='<div class="zrow'+(on?" on":"")+'"'+(on?' style="background:'+bc.bg+';color:'+bc.fg+'"':'')+'>'

          +'<span style="text-transform:capitalize">'+tab.zones[i]+'</span><span>'+fmtPrice(th)+'</span></div>';

    });

  }

  let extra="";

  const ex=(hasZones&&idx!=null)?zsch.extras(idx,s.t,tab.zones):null;

  if(ex){

    if(ex.type==="extra") extra='<div class="zextra" style="background:'+COLORS[ex.color].bg+'">'+ex.text+'</div>';

    else extra='<div class="zhint">'+ex.text+'</div>';

  }

  if(s.row8!=null){

    extra+='<div class="zhint">🧠 ที่ '+fmtPrice(s.row8)+'</div>';

  }

  if(s.row9!=null){

    extra+='<div class="zhint">🧀 ที่ '+fmtPrice(s.row9)+'</div>';

  }

  const noteLine = s.note ? '<div class="d-note">'+(tab.line2==="dca"?"DCA เดือนนี้: ฿":"")+esc(s.note)+'</div>' : "";

  const eyes = (s.row8!=null && p.value!=null && p.value <= s.row8) ? " 🧠" : "";

  const cheese = (s.row9!=null && p.value!=null && p.value >= s.row9) ? " 🧀" : "";

  document.getElementById("sheet").innerHTML=

    '<div class="d-head"><div class="d-chip" style="background:'+c.bg+'"></div>'

    +'<div><div class="d-tk">'+esc(s.ticker)+eyes+cheese+'</div>'+noteLine+'</div>'

    +'<div class="d-price">'+fmtPrice(p.value)

    +'<div class="d-src">'+(p.live?"● ราคาสด (Finnhub)":"ราคาจาก Sheet (~20 นาที)")+'</div></div></div>'

    +rows+extra;

  document.getElementById("overlay").classList.add("show");

}

document.getElementById("overlay").onclick=function(e){

};

document.getElementById("refresh").onclick=loadAll;


// auto-refresh ราคา: วิ่งเองเมื่อหน้า active และครบรอบเวลา

async function autoTick(){

  if(document.visibilityState!=="visible") return;

  if(Date.now()-state.lastQuoteFetch<AUTO_REFRESH_MS) return;

  const btn=document.getElementById("refresh");

  btn.classList.add("spin");

  await fetchAllQuotes();

  render();

  btn.classList.remove("spin");

}

setInterval(autoTick, 15*1000);

document.addEventListener("visibilitychange", ()=>{

  if(document.visibilityState==="visible") autoTick();

});
